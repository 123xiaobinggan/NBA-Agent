CREATE ALGORITHM = UNDEFINED DEFINER = `root` @`localhost` SQL SECURITY DEFINER VIEW `v_player_season_overall` AS
select
    `pss`.`stat_id` AS `stat_id`,
    `pss`.`season_id` AS `season_id`,
    `s`.`season_label` AS `season_label`,
    `pss`.`season_type` AS `season_type`,
    `pss`.`player_id` AS `player_id`,
    `p`.`full_name` AS `player_name`,
    `pss`.`position` AS `position`,
    `pss`.`age` AS `age`,
    `pss`.`games_played` AS `games_played`,
    `pss`.`games_started` AS `games_started`,
    `pss`.`minutes_per_game` AS `minutes_per_game`,
    `pss`.`field_goal_pct` AS `field_goal_pct`,
    `pss`.`three_point_pct` AS `three_point_pct`,
    `pss`.`free_throw_pct` AS `free_throw_pct`,
    `pss`.`rebounds_per_game` AS `rebounds_per_game`,
    `pss`.`assists_per_game` AS `assists_per_game`,
    `pss`.`steals_per_game` AS `steals_per_game`,
    `pss`.`blocks_per_game` AS `blocks_per_game`,
    `pss`.`turnovers_per_game` AS `turnovers_per_game`,
    `pss`.`points_per_game` AS `points_per_game`
from (
        (
            `player_season_stats` `pss`
            join `players` `p` on (
                (
                    `p`.`player_id` = `pss`.`player_id`
                )
            )
        )
        join `seasons` `s` on (
            (
                `s`.`season_id` = `pss`.`season_id`
            )
        )
    )
where (
        (`pss`.`is_total` = true)
        or (
            (`pss`.`is_total` = false)
            and (
                (
                    select count(0)
                    from `player_season_stats` `x`
                    where (
                            (
                                `x`.`season_id` = `pss`.`season_id`
                            )
                            and (
                                `x`.`season_type` = `pss`.`season_type`
                            )
                            and (
                                `x`.`player_id` = `pss`.`player_id`
                            )
                        )
                ) = 1
            )
        )
    )

    `tgs`.`second_chance_points` AS `second_chance_points`,
    `tgs`.`fast_break_points` AS `fast_break_points`,
    `tgs`.`largest_lead` AS `largest_lead`,
    `tgs`.`points_off_turnovers` AS `points_off_turnovers`
from (
        (
            (
                (
                    `team_game_stats` `tgs`
                    join `games` `g` on (
                        (
                            `g`.`game_id` = `tgs`.`game_id`
                        )
                    )
                )
                join `seasons` `s` on (
                    (
                        `s`.`season_id` = `g`.`season_id`
                    )
                )
            )
            join `teams` `t` on (
                (
                    `t`.`team_id` = `tgs`.`team_id`
                )
            )
        )
        join `teams` `ot` on (
            (
                `ot`.`team_id` = `tgs`.`opponent_team_id`
            )
        )
    )
